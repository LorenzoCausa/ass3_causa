230 {

270 template<typename T>
271 bool LevMarq<T>::step(F_z_x f_z_x, F_z_J f_J)
272 {
273  f_J(curr_z, _J);
274  Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> Jt = _J.transpose();
275  Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> JtJ = (Jt * _J);
276 
277  eVector B = -Jt * x64;
278  if (mu < 0)
279  {
280  // first time only
281  int max = 0;
282  for (int j = 1; j < JtJ.cols(); j++)
283  if (JtJ(j, j) > JtJ(max, max))
284  max = j;
285  mu = JtJ(max, max) * _tau;
286  }
287 
288  double gain = 0, prev_mu = 0;
289  int ntries = 0;
290  bool isStepAccepted = false;
291  do
292  {
293  // add/update dumping factor to JtJ.
294  // very efficient in any case, but particularly if initial dump does not produce improvement and must reenter
295  for (int j = 0; j < JtJ.cols(); j++)
296  JtJ(j, j) += mu - prev_mu; // update mu
297  prev_mu = mu;
298  eVector delta = JtJ.ldlt().solve(B);
299  eVector estimated_z = curr_z + delta;
300 
301  // compute error
302  f_z_x(estimated_z, x64);
303  auto err = x64.cwiseProduct(x64).sum();
304  auto L = 0.5 * delta.transpose() * ((mu * delta) - B);
305  gain = (err - prevErr) / L(0, 0);
306 
307  // get gain
308  if (gain > 0)
309  {
310  mu = mu * std::max(double(0.33), 1. - pow(2 * gain - 1, 3));
311  v = 5.f;
312  currErr = err;
313  curr_z = estimated_z;
314  isStepAccepted = true;
315  }
316  else
317  {
318  mu = mu * v;
319  v = v * 5;
320  }
321  } while (gain <= 0 && ntries++ < 5);
322 
323  if (_verbose)
324  std::cout << std::setprecision(5) << "Curr Error=" << currErr << " AErr(prev - curr) = " << prevErr - currErr
325  << " gain = " << gain << " dumping factor = " << mu << std::endl;
326  // check if we must move to the new position or exit
327  if (currErr < prevErr)
328  std::swap(currErr, prevErr);
329 
330  return isStepAccepted;
331 
332 }

342 double LevMarq<T>::solve(eVector &z, F_z_x f_z_x, F_z_J f_J)
343 {
344  init(z, f_z_x);
345 
346  if (_stopFunction)
347  {
348  do
349  {
350  step(f_z_x, f_J);
351  if (_step_callback)
352  _step_callback(curr_z);
353  } while (!_stopFunction(curr_z));
354  }
355  else
356  {
357  // Initial error estimation
358  int mustExit = 0;
359  for (int i = 0; i < _maxIters && !mustExit; i++)
360  {
361  if (_verbose)
362  std::cerr << "iteration " << i << "/" << _maxIters << " ";
363  bool isStepAccepted = step(f_z_x, f_J);
364 
365  // check if we must exit
366  if (currErr < _minErrorAllowed)
367  mustExit = 1;
368  if (fabs(prevErr - currErr) <= _min_step_error_diff || !isStepAccepted)
369  mustExit = 2;
370 
371  // exit if error increment
372  if (currErr < prevErr)
373  mustExit = 3;
374 // if ((prevErr - currErr) < 1e-5)
375 // mustExit = true;
376  if (_step_callback)
377  _step_callback(curr_z);
378  }

381  }
